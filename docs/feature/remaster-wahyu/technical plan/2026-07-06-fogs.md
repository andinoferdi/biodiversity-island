# Fog Toggle and POV Persistence Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking. Do not use subagents for this small follow-up.

**Goal:** Add a Fog On/Off feature with default On, keep fog visible in Low graphics, and fix the bug where fog visually disappears when entering animal POV.

**Architecture:** Keep the existing scene-level `<fog attach="fog">` in `Sky.tsx`, but give it its own `isFoggy` target blend instead of tying visibility only to rain. Add a lightweight world-space fog layer that remains visible from both orthographic overview and ground-level POV; this solves the camera-distance root cause without changing animal controls or terrain. Wire a new UI toggle from `BiodiversityPrototype.tsx` through `IslandScene.tsx` into `Sky` and the new fog layer.

**Tech Stack:** Next.js 16, React 19, TypeScript, Three.js 0.185, @react-three/fiber v9, Tailwind CSS v4, npm.

## Root Cause Investigation

Current fog is only scene-level Three.js fog in `src/components/prototype/effects/Sky.tsx`:

```tsx
<fog ref={fogRef} attach="fog" args={[PALETTE.fogClear, FOG_NEAR, FOG_FAR]} />
```

The color animates with rain:

```ts
if (fogRef.current) {
  fogRef.current.color.lerpColors(colors.fogClear, colors.fogRain, blend);
}
```

Problems:

1. There is no `isFoggy` state and no Fog On/Off UI. Fog is always attached and only changes color with rain.
2. Three.js linear scene fog is camera-distance based. From the orthographic overview, terrain is far enough from the camera for `FOG_NEAR = 26` / `FOG_FAR = 72` to show haze. In animal POV, the perspective camera sits near the animal and nearby terrain/vegetation are inside the clear zone before `FOG_NEAR`, so fog appears to vanish.
3. Low graphics currently disables postprocessing and several shadow/fill features, but fog should not be treated as a high-quality effect. The plan must keep fog active in Low unless the Fog toggle itself is Off.
4. Potential hidden bug: if fog visibility is implemented by conditionally unmounting `<Sky>` or the fog object, rain color transitions and scene background could pop. The fix should use `useEasedBlend`, mirroring Rain/Cloud behavior.

Design decision: scene fog remains for long-distance atmospheric depth, and a new low-cost `FogLayer` provides visible low-altitude haze in POV. The `FogLayer` is always available in every graphic quality tier and uses deterministic placement, transparent planes, and ref-based animation.

## Task 3: Add World-Space FogLayer for POV Visibility

**Files:**
- Create: `src/components/prototype/effects/FogLayer.tsx`
- Modify: `src/components/prototype/IslandScene.tsx`

**Interface:**
- Props: `{ isFoggy: boolean; isRaining: boolean; timeScale: TimeScale }`.
- Always rendered regardless of `graphicQuality`.
- Uses transparent, deterministic, low-cost fog banks so Low graphics still has fog.

- [ ] **Step 1: Create `FogLayer.tsx`**

Create `src/components/prototype/effects/FogLayer.tsx` with:

```tsx
"use client";

import { useMemo, useRef } from "react";
import { useFrame } from "@react-three/fiber";
import * as THREE from "three";
import type { TimeScale } from "../simulation";
import { PALETTE, seededRandom, useEasedBlend, WIND } from "./weather";

const FOG_BANK_COUNT = 18;
const AREA_HALF = 12;
const LAYOUT_SEED = 20260707;
const BASE_Y = 0.72;
const Y_VARIATION = 0.28;
const BASE_OPACITY = 0.16;
const RAIN_OPACITY_BOOST = 0.08;
const BANK_SCALE_MIN = 3.8;
const BANK_SCALE_MAX = 7.2;
const DRIFT_SPEED = 0.12;
const HIDDEN_Y = -10;

interface FogBank {
  x: number;
  z: number;
  y: number;
  scaleX: number;
  scaleZ: number;
  rotY: number;
  driftScale: number;
}

function makeFogTexture(): THREE.CanvasTexture {
  const canvas = document.createElement("canvas");
  canvas.width = 128;
  canvas.height = 128;
  const ctx = canvas.getContext("2d")!;
  const g = ctx.createRadialGradient(64, 64, 0, 64, 64, 64);
  g.addColorStop(0, "rgba(255,255,255,0.75)");
  g.addColorStop(0.45, "rgba(255,255,255,0.38)");
  g.addColorStop(0.78, "rgba(255,255,255,0.10)");
  g.addColorStop(1, "rgba(255,255,255,0)");
  ctx.fillStyle = g;
  ctx.fillRect(0, 0, 128, 128);
  const texture = new THREE.CanvasTexture(canvas);
  texture.colorSpace = THREE.SRGBColorSpace;
  return texture;
}

function buildFogBanks(): FogBank[] {
  const rand = seededRandom(LAYOUT_SEED);
  return Array.from({ length: FOG_BANK_COUNT }, () => {
    const angle = rand() * Math.PI * 2;
    const radius = Math.sqrt(rand()) * AREA_HALF;
    const scale = BANK_SCALE_MIN + rand() * (BANK_SCALE_MAX - BANK_SCALE_MIN);
    return {
      x: Math.cos(angle) * radius,
      z: Math.sin(angle) * radius,
      y: BASE_Y + (rand() - 0.5) * Y_VARIATION,
      scaleX: scale * (1.2 + rand() * 0.9),
      scaleZ: scale * (0.45 + rand() * 0.35),
      rotY: rand() * Math.PI,
      driftScale: 0.65 + rand() * 0.7,
    };
  });
}

interface FogLayerProps {
  isFoggy: boolean;
  isRaining: boolean;
  timeScale: TimeScale;
}

export default function FogLayer({ isFoggy, isRaining, timeScale }: FogLayerProps) {
  const meshRef = useRef<THREE.InstancedMesh>(null);
  const fogBlend = useEasedBlend(isFoggy);
  const rainBlend = useEasedBlend(isRaining);
  const banks = useMemo(() => buildFogBanks(), []);
  const texture = useMemo(() => makeFogTexture(), []);
  const dummy = useMemo(() => new THREE.Object3D(), []);
  const centers = useRef(banks.map((bank) => ({ x: bank.x, z: bank.z })));
  const colors = useMemo(
    () => ({
      clear: new THREE.Color(PALETTE.fogClear),
      rain: new THREE.Color(PALETTE.fogRain),
    }),
    [],
  );

  useFrame((_, delta) => {
    const mesh = meshRef.current;
    if (!mesh) return;

    const fogPresence = fogBlend.current;
    const storm = rainBlend.current;
    const material = mesh.material as THREE.MeshBasicMaterial;
    material.color.lerpColors(colors.clear, colors.rain, storm);
    material.opacity = (BASE_OPACITY + storm * RAIN_OPACITY_BOOST) * fogPresence;
    mesh.visible = fogPresence > 0.01;

    const move = delta * timeScale;
    for (let i = 0; i < banks.length; i++) {
      const bank = banks[i];
      const center = centers.current[i];
      center.x += WIND.x * DRIFT_SPEED * bank.driftScale * move;
      center.z += WIND.z * DRIFT_SPEED * bank.driftScale * move;
      if (center.x < -AREA_HALF) center.x += AREA_HALF * 2;
      if (center.x > AREA_HALF) center.x -= AREA_HALF * 2;
      if (center.z < -AREA_HALF) center.z += AREA_HALF * 2;
      if (center.z > AREA_HALF) center.z -= AREA_HALF * 2;

      dummy.position.set(center.x, fogPresence > 0.01 ? bank.y : HIDDEN_Y, center.z);
      dummy.rotation.set(-Math.PI / 2, 0, bank.rotY);
      dummy.scale.set(bank.scaleX, bank.scaleZ, 1);
      dummy.updateMatrix();
      mesh.setMatrixAt(i, dummy.matrix);
    }
    mesh.instanceMatrix.needsUpdate = true;
  });

  return (
    <instancedMesh ref={meshRef} args={[undefined, undefined, FOG_BANK_COUNT]}>
      <planeGeometry args={[1, 1]} />
      <meshBasicMaterial
        map={texture}
        transparent
        opacity={BASE_OPACITY}
        depthWrite={false}
        side={THREE.DoubleSide}
      />
    </instancedMesh>
  );
}
```

Rationale:
- Horizontal transparent fog banks remain visible from top-down and ground-level POV.
- `depthWrite={false}` prevents fog planes from cutting holes into terrain/animals.
- Component is cheap enough for Low: 18 instanced planes, no postprocessing.
